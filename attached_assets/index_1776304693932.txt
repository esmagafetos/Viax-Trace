function setFile(file) {
    const ext = file.name.split('.').pop().toLowerCase();
    if (!['xlsx','csv'].includes(ext)) { showToast('Formato inválido. Use .xlsx ou .csv'); return; }
    if (file.size > 10 * 1024 * 1024) { showToast('Arquivo muito grande. Máximo 10MB.'); return; }
    currentFile = file;
    fileStripName.textContent = file.name;
    fileStripMeta.textContent = formatBytes(file.size);
    fileStrip.classList.add('visible');
    processBtn.disabled = false;
}

function clearFile() {
    currentFile = null;
    fileInput.value = '';
    fileStrip.classList.remove('visible');
    processBtn.disabled = true;
}

// ── PROCESS COM SSE (FETCH + STREAM) ──────────────────────────────────────
processBtn.addEventListener('click', async () => {
    if (!currentFile) return;

    // UI de loading
    processBtn.style.display = 'none';
    loadingWrap.classList.add('visible');
    loadingSteps.innerHTML = '';
    resultsArea.classList.remove('visible');
    loadingTxt.textContent = 'Conectando ao servidor...';

    abortController = new AbortController();

    try {
        const formData = new FormData();
        formData.append('arquivo', currentFile);

        const response = await fetch('api.php', {
            method: 'POST',
            body: formData,
            signal: abortController.signal
        });

        const reader = response.body.getReader();
        const decoder = new TextDecoder();
        let buffer = '';

        while (true) {
            const { done, value } = await reader.read();
            if (done) break;

            buffer += decoder.decode(value, { stream: true });
            const lines = buffer.split('\n');
            buffer = lines.pop() || '';

            for (let i = 0; i < lines.length; i++) {
                const line = lines[i];
                if (line.startsWith('event: ')) {
                    const eventType = line.slice(7);
                    // Aguarda a próxima linha que contém data:
                    const nextLine = lines[i+1];
                    if (nextLine && nextLine.startsWith('data: ')) {
                        const dataStr = nextLine.slice(6);
                        try {
                            const data = JSON.parse(dataStr);
                            if (eventType === 'step') {
                                addLoadingStep(data.step);
                                loadingTxt.textContent = data.step;
                            } else if (eventType === 'result') {
                                currentData = data.result;
                                renderResults(data.result);
                            } else if (eventType === 'error') {
                                throw new Error(data.error || 'Erro no servidor');
                            }
                        } catch (e) {
                            console.warn('Erro ao parsear evento', e);
                        }
                        i++; // pula a linha de data já processada
                    }
                }
            }
        }
    } catch (err) {
        if (err.name === 'AbortError') {
            showToast('Processamento cancelado pelo usuário.');
        } else {
            showToast('Erro: ' + err.message);
        }
    } finally {
        loadingWrap.classList.remove('visible');
        processBtn.style.display = '';
        abortController = null;
    }
});

function addLoadingStep(text) {
    const div = document.createElement('div');
    div.className = 'loading-step';
    div.innerHTML = `<div class="step-dot"></div><span>${escapeHtml(text)}</span>`;
    loadingSteps.appendChild(div);
    // Scroll automático para o último passo
    div.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
}

// ── RENDER RESULTS ─────────────────────────────────────────────────────────
function renderResults(data) {
    const { total_enderecos, total_nuances, percentual_problema, detalhes, metricas_tecnicas } = data;

    statsRow.innerHTML = `
        <div class="stat-tile">
            <div class="stat-val">${total_enderecos}</div>
            <div class="stat-lbl">Analisados</div>
        </div>
        <div class="stat-tile accent-line">
            <div class="stat-val danger">${total_nuances}</div>
            <div class="stat-lbl">Nuances</div>
        </div>
        <div class="stat-tile accent-line">
            <div class="stat-val danger">${percentual_problema}%</div>
            <div class="stat-lbl">Com problema</div>
        </div>
        <div class="stat-tile ok-line">
            <div class="stat-val good">${metricas_tecnicas.taxa_geocode_sucesso}%</div>
            <div class="stat-lbl">Geocode ok</div>
        </div>
    `;

    renderTable(detalhes, activeFilter);

    metricsList.innerHTML = [
        { icon: clockIcon(), label: 'Tempo de processamento', val: metricas_tecnicas.tempo_processamento_ms + ' ms' },
        { icon: dbIcon(), label: 'Cache hits', val: metricas_tecnicas.cache_hits },
        { icon: mapIcon(), label: 'Geocodificação com sucesso', val: metricas_tecnicas.taxa_geocode_sucesso + '%' },
        { icon: shieldIcon(), label: 'Rate limit', val: metricas_tecnicas.limite_requisicoes },
    ].map(m => `
        <div class="metric-row">
            <span class="metric-key">${m.icon} ${m.label}</span>
            <span class="metric-val">${m.val}</span>
        </div>
    `).join('');

    renderChart(total_enderecos - total_nuances, total_nuances);
    resultsArea.classList.add('visible');
}

function renderTable(detalhes, filter) {
    const filtered = filter === 'all' ? detalhes
        : filter === 'nuance' ? detalhes.filter(d => d.is_nuance)
        : detalhes.filter(d => !d.is_nuance);

    if (!filtered.length) {
        tableBody.innerHTML = '';
        emptyState.style.display = '';
        return;
    }
    emptyState.style.display = 'none';

    tableBody.innerHTML = filtered.map(item => {
        const sim = item.similaridade !== null ? item.similaridade : null;
        const simPct = sim !== null ? Math.round(sim * 100) : null;
        const simBar = sim !== null
            ? `<div class="sim-bar-wrap">
                 <div class="sim-bar"><div class="sim-fill${simPct < 68 ? ' low' : ''}" style="width:${simPct}%"></div></div>
                 <span class="sim-val">${simPct}%</span>
               </div>`
            : '<span class="sim-val">N/A</span>';

        const badge = item.is_nuance
            ? `<span class="badge badge-nuance">${warnIcon()} Nuance</span>`
            : `<span class="badge badge-ok">${checkIcon()} OK</span>`;

        const motivo = item.motivo
            ? `<span class="motivo-cell" title="${escapeHtml(item.motivo)}">${escapeHtml(item.motivo)}</span>`
            : '<span style="color:var(--text-faint)">—</span>';

        return `<tr>
            <td class="col-line">${item.linha}</td>
            <td class="col-orig"><span title="${escapeHtml(item.endereco_original)}">${escapeHtml(trunc(item.endereco_original, 55))}</span></td>
            <td class="col-name">${escapeHtml(item.nome_rua_extraido) || '<span style="color:var(--text-faint)">—</span>'}</td>
            <td class="col-official">${escapeHtml(item.nome_rua_oficial) || '<em style="color:var(--text-faint)">não encontrado</em>'}</td>
            <td class="col-sim">${simBar}</td>
            <td>${badge}</td>
            <td>${motivo}</td>
        </tr>`;
    }).join('');
}

// ── CHART ──────────────────────────────────────────────────────────────────
function getChartColors() {
    const dark = document.documentElement.getAttribute('data-theme') === 'dark';
    return {
        ok: dark ? '#2ea863' : '#1a7a4a',
        nuance: dark ? '#e8703a' : '#d4521a',
        text: dark ? '#8a877f' : '#6b6860',
    };
}

function renderChart(okCount, nuanceCount) {
    const ctx = document.getElementById('nuanceChart').getContext('2d');
    if (chartInstance) chartInstance.destroy();
    const c = getChartColors();
    chartInstance = new Chart(ctx, {
        type: 'doughnut',
        data: {
            labels: ['OK', 'Nuance'],
            datasets: [{
                data: [okCount, nuanceCount],
                backgroundColor: [c.ok, c.nuance],
                borderWidth: 0,
                hoverOffset: 4,
            }]
        },
        options: {
            responsive: false,
            cutout: '68%',
            plugins: {
                legend: {
                    position: 'bottom',
                    labels: {
                        color: c.text,
                        font: { size: 11, family: "'Poppins', sans-serif" },
                        boxWidth: 10,
                        padding: 12,
                    }
                }
            }
        }
    });
}

function updateChartTheme() {
    if (!chartInstance || !currentData) return;
    const c = getChartColors();
    chartInstance.data.datasets[0].backgroundColor = [c.ok, c.nuance];
    chartInstance.options.plugins.legend.labels.color = c.text;
    chartInstance.update();
}

// ── EXPORT ────────────────────────────────────────────────────────────────
exportBtn.addEventListener('click', () => {
    if (!currentData) return;
    const rows = [['Linha','Endereço Original','Rua Extraída','Rua Oficial','Similaridade','Is Nuance','Motivo']];
    currentData.detalhes.forEach(d => {
        rows.push([d.linha, d.endereco_original, d.nome_rua_extraido||'', d.nome_rua_oficial||'',
            d.similaridade !== null ? d.similaridade : '', d.is_nuance ? 'Sim' : 'Não', d.motivo||'']);
    });
    const csv = rows.map(r => r.map(c => `"${String(c).replace(/"/g,'""')}"`).join(',')).join('\n');
    const blob = new Blob(['\uFEFF' + csv], { type: 'text/csv;charset=utf-8;' });
    const url = URL.createObjectURL(blob);
    const a = document.createElement('a');
    a.href = url; a.download = 'viax-scout-relatorio.csv';
    document.body.appendChild(a); a.click();
    document.body.removeChild(a); URL.revokeObjectURL(url);
});

// ── TOAST ─────────────────────────────────────────────────────────────────
let toastTimer;
function showToast(msg) {
    toast.textContent = msg;
    toast.classList.add('visible');
    clearTimeout(toastTimer);
    toastTimer = setTimeout(() => toast.classList.remove('visible'), 4000);
}

// ── UTILS ─────────────────────────────────────────────────────────────────
function escapeHtml(s) {
    if (!s) return '';
    return s.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}
function trunc(s, n) { return s && s.length > n ? s.slice(0, n) + '…' : (s || ''); }
function formatBytes(b) {
    if (b < 1024) return b + ' B';
    if (b < 1024*1024) return (b/1024).toFixed(1) + ' KB';
    return (b/(1024*1024)).toFixed(1) + ' MB';
}
